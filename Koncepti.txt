25. Rekurzije je najlakše koristiti primjenom rekurzivnih algoritama sortiranja kao što su quick sort,
merge sort, insert sort ili pretraživanja kao što je binary search.

	int rekurzivna_binarna_pretraga(PODATAK_NAMIRNICA* polje, int lijevo, int desno, int kljuc)
	{
    	int sredina;

    	if (desno >= lijevo)
    	{
        	sredina = lijevo + (desno - lijevo) / 2;

        	if (polje[sredina].identifikator == kljuc)
        	{
            	return sredina;
        	}

        	if (polje[sredina].identifikator > kljuc)
        	{
            	return rekurzivna_binarna_pretraga(polje, lijevo, sredina - 1, kljuc);
        	}
        	else
        	{
            	return rekurzivna_binarna_pretraga(polje, sredina + 1, desno, kljuc);
        	}
    	}

    	return -1;
	}
